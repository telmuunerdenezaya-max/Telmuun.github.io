#include <stdio.h>
#include <stdlib.h>

void read(int *data, int size);
void max_min(int *data, int size, int *max, int *min);
int find(int *data, int size, int x);

int main() {
    int N;
    printf("Элементийн тоо N = ");
    scanf("%d", &N);

    int *data = (int *)malloc(N * sizeof(int));
    if (data == NULL) {
        printf("Санах ой нөөцлөгдөхөд алдаа гарлаа!\n");
        return 1;
    }
    
    read(data, N);

    int max, min;
    max_min(data, N, &max, &min);
    printf("Хамгийн их утга: %d\n", max);
    printf("Хамгийн бага утга: %d\n", min);

    int x;
    printf("Хайх утга x = ");
    scanf("%d", &x);

    if (find(data, N, x))
        printf("%d утга олдлоо.\n", x);
    else
        printf("%d утга олдсонгүй.\n", x);

    free(data);
    return 0;
}

void read(int *data, int size) {
    printf("%d ширхэг тоо оруулна уу:\n", size);
    for (int i = 0; i < size; i++) {
        printf("data[%d] = ", i);
        scanf("%d", &data[i]);
    }
}

void max_min(int *data, int size, int *max, int *min) {
    *max = *min = data[0];
    for (int i = 1; i < size; i++) {
        if (data[i] > *max)
            *max = data[i];
        if (data[i] < *min)
            *min = data[i];
    }
}

int find(int *data, int size, int x) {
    for (int i = 0; i < size; i++) {
        if (data[i] == x)
            return 1; 
    }
    return 0;
}
